const dateformat = (date, format) => {
  var z = {
    y: date.getFullYear(), 
    M: date.getMonth() + 1,
    d: date.getDate(), 
    h: date.getHours(),
    m: date.getMinutes(), 
    s: date.getSeconds() 
  };
  return format.replace(/(y+|M+|d+|h+|m+|s+)/g, function (v) {
    return ((v.length > 1 ? "0" : "") + eval('z.' + v.slice(-1))).slice(-(v.length > 2 ? v.length : 2)) 
  });
}
