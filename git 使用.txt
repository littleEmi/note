git 基本命令
  git clone https://github.com*****.
  git pull 拉取当前最新代码
  修改内容后
  git add .  上传当前目录
  git commit -m "提交当前代码注释"
  git push  提交
  
  设置git用户名密码(避免多次输入)
  git config --global user.name "用户名"
  git config --global user.email 邮箱

  分支
   git branch 分支名
   git checkout 分支名  切换到分支
   git push --set-upstream origin 版本名
